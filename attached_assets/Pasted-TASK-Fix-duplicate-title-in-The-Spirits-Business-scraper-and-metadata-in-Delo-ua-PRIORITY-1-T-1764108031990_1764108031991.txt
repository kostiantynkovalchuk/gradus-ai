Add _clean_content method (similar to other scrapers):
```python
def _clean_content(self, content: str, title: str) -> str:
    """Clean The Spirits Business content"""
    if not content:
        return ""
    
    import re
    
    # Remove duplicate title at start
    # Check if title appears twice
    title_pattern = re.escape(title)
    double_title_pattern = f'^{title_pattern}\\s*\\n\\s*{title_pattern}'
    
    if re.match(double_title_pattern, content, re.MULTILINE):
        # Remove first occurrence
        content = re.sub(f'^{title_pattern}\\s*\\n', '', content, count=1, flags=re.MULTILINE)
    
    # Get lines
    lines = [l.strip() for l in content.split('\n') if l.strip()]
    
    # Remove metadata lines (author, date, etc)
    filtered = []
    for line in lines:
        # Skip very short lines that might be metadata
        if len(line) < 20 and not line.endswith('.'):
            continue
        # Skip date patterns
        if re.match(r'\d{1,2}\s+\w+,?\s+\d{4}', line):
            continue
        filtered.append(line)
    
    # Join text
    text = ' '.join(filtered)
    text = re.sub(r'  +', ' ', text)
    
    # Add paragraph breaks
    text = re.sub(r'([.!?»"]\s+)([A-ZА-ЯІЇЄ])', r'\1\n\n\2', text)
    text = re.sub(r'\n{3,}', '\n\n', text)
    
    return text.strip()
```

```python
def _clean_content(self, content: str, title: str) -> str:
    """Clean Delo.ua content with image caption filtering"""
    if not content:
        return ""
    
    import re
    
    # Remove "Новині компаній"
    content = re.sub(r'Новині компаній', '', content)
    
    # EXPLICIT duplicate title removal
    title_pattern = re.escape(title)
    double_title_pattern = f'^{title_pattern}\\s*\\n\\s*{title_pattern}'
    if re.match(double_title_pattern, content, re.MULTILINE):
        content = re.sub(f'^{title_pattern}\\s*\\n', '', content, count=1, flags=re.MULTILINE)
    
    # Get lines
    lines = [l.strip() for l in content.split('\n') if l.strip()]
    
    # Filter lines
    filtered = []
    for line in lines:
        # Skip metadata
        if line in ['Категорія', 'Новини', 'Дата публікації']:
            continue
        
        # Skip image captions (pattern: "Text / Depositphotos" or "Text / Getty Images")
        if ' / ' in line and len(line) < 100:
            # Check if it's photo credit
            parts = line.split(' / ')
            if len(parts) == 2 and parts[1].strip() in ['Depositphotos', 'Getty Images', 'Reuters', 'AFP', 'Unsplash']:
                continue
        
        filtered.append(line)
    
    # Join text
    text = ' '.join(filtered)
    text = re.sub(r'  +', ' ', text)
    
    # Add paragraph breaks
    text = re.sub(r'([.!?»"]\s+)([A-ZА-ЯІЇЄ])', r'\1\n\n\2', text)
    text = re.sub(r'\n{3,}', '\n\n', text)
    
    return text.strip()
```
