4. UPDATED CLEANING FUNCTION:
   Add this method to DeloUaScraper class:
```python
   def _clean_content(self, content: str, title: str) -> str:
       """Clean and format article content"""
       if not content:
           return ""
       
       # Remove duplicate title if present at start
       if content.startswith(title):
           content = content[len(title):].strip()
       
       # Remove metadata patterns
       metadata_patterns = [
           r'Категорія\s*\n\s*[^\n]+\n',
           r'Дата публікації\s*\n\s*[^\n]+\n',
           r'Новини\s*\n',
           r'^\d{1,2}\s+\w+\s+\d{2}:\d{2}\s*\n'
       ]
       
       import re
       for pattern in metadata_patterns:
           content = re.sub(pattern, '', content, flags=re.MULTILINE)
       
       # Clean up whitespace
       lines = content.split('\n')
       cleaned_lines = []
       
       for line in lines:
           line = line.strip()
           if line and line not in ['Категорія', 'Новини', 'Дата публікації']:
               cleaned_lines.append(line)
       
       # Join with double newline for paragraph separation
       content = '\n\n'.join(cleaned_lines)
       
       # Remove multiple consecutive newlines
       content = re.sub(r'\n{3,}', '\n\n', content)
       
       return content.strip()
```
