```python
   @app.get("/api/debug/article/{article_id}")
   async def debug_article(article_id: int, db: Session = Depends(get_db)):
       article = db.query(ContentQueue).filter(ContentQueue.id == article_id).first()
       if not article:
           return {"error": "Article not found"}
       
       # Check why it might not appear on website
       issues = []
       if not article.translated_title:
           issues.append("Missing translated_title")
       if not article.translated_text:
           issues.append("Missing translated_text")
       if not article.image_data and not article.image_url:
           issues.append("Missing image")
       if not article.posted_at:
           issues.append("Missing posted_at timestamp")
       if article.status != 'posted':
           issues.append(f"Status is '{article.status}', not 'posted'")
       
       return {
           "id": article.id,
           "title": article.translated_title,
           "status": article.status,
           "posted_at": str(article.posted_at),
           "has_image": bool(article.image_data or article.image_url),
           "has_content": bool(article.translated_text),
           "potential_issues": issues,
           "should_appear_on_website": len(issues) == 0
       }
```
