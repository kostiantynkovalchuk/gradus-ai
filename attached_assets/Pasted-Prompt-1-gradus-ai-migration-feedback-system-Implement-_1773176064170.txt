async def handle_solomon_feedback(callback: CallbackQuery):
    parts = callback.data.split("_", 3)
    # sfb_like_DOCID_CASENUMBER or sfb_dislike_DOCID_CASENUMBER
    feedback = parts[1]  # "like" or "dislike"
    doc_id = parts[2]
    cause_number = parts[3].replace("_", "/") if len(parts) > 3 else ""
    
    user_id = callback.from_user.id
    
    # Get session_id from DB
    conn = get_db_connection()
    cur = conn.cursor()
    cur.execute("SELECT id FROM solomon_sessions WHERE telegram_id = %s", (user_id,))
    session = cur.fetchone()
    session_id = session[0] if session else None
    
    # Save feedback
    cur.execute("""
        INSERT INTO solomon_feedback (session_id, doc_id, cause_number, feedback)
        VALUES (%s, %s, %s, %s)
        ON CONFLICT DO NOTHING
    """, (session_id, doc_id, cause_number, feedback))
    conn.commit()
    cur.close()
    conn.close()
    
    # Update button to show selection
    if feedback == "like":
        new_text = "✅ Корисно"
        reply = "Дякую! Це допоможе покращити пошук."
    else:
        new_text = "❌ Не те"
        reply = "Зрозуміло. Спробуйте уточнити запит."
    
    # Update keyboard — mark selected button
    try:
        await callback.message.edit_reply_markup(reply_markup=InlineKeyboardMarkup(inline_keyboard=[[
            InlineKeyboardButton(text=new_text if feedback=="like" else "👍 Корисно", callback_data="sfb_done"),
            InlineKeyboardButton(text=new_text if feedback=="dislike" else "👎 Не те", callback_data="sfb_done"),
            InlineKeyboardButton(text="🔗 Читати", url=f"https://reyestr.court.gov.ua/Review/{doc_id}"),
        ]]))
    except:
        pass
    
    await callback.answer(reply)

async def solomon_analytics_data(credentials: HTTPBasicCredentials = Depends(verify_solomon_credentials)):
    conn = get_db_connection()
    cur = conn.cursor()
    
    cur.execute("""
        SELECT 
            COUNT(*) FILTER (WHERE feedback='like') as likes,
            COUNT(*) FILTER (WHERE feedback='dislike') as dislikes,
            COUNT(DISTINCT session_id) as users,
            COUNT(*) as total
        FROM solomon_feedback
        WHERE created_at > NOW() - INTERVAL '30 days'
    """)
    stats = cur.fetchone()
    
    cur.execute("""
        SELECT cause_number, COUNT(*) as likes
        FROM solomon_feedback
        WHERE feedback = 'like'
        GROUP BY cause_number
        ORDER BY likes DESC
        LIMIT 5
    """)
    top_liked = cur.fetchall()
    
    cur.execute("""
        SELECT cause_number, COUNT(*) as dislikes
        FROM solomon_feedback  
        WHERE feedback = 'dislike'
        GROUP BY cause_number
        ORDER BY dislikes DESC
        LIMIT 5
    """)
    top_disliked = cur.fetchall()
    
    cur.close()
    conn.close()
    
    return {
        "last_30_days": {
            "likes": stats[0], "dislikes": stats[1],
            "users": stats[2], "total": stats[3],
            "satisfaction": round(stats[0] / stats[3] * 100) if stats[3] > 0 else 0
        },
        "top_liked": [{"cause_number": r[0], "likes": r[1]} for r in top_liked],
        "top_disliked": [{"cause_number": r[0], "dislikes": r[1]} for r in top_disliked],
    }
