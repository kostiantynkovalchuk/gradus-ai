async def handle_hunt_action(callback_query: dict, db):
    data = callback_query["data"]  
    # parse: hunt_action_{action}_{vacancy_id}
    parts = data.split("_")
    action = parts[2]  # search/post/both/skip
    vacancy_id = int(parts[3])
    
    chat_id = callback_query["message"]["chat"]["id"]
    message_id = callback_query["message"]["message_id"]
    thread_id = callback_query["message"].get("message_thread_id")
    
    # Answer callback to remove spinner
    await answer_callback_query(callback_query["id"])
    
    status_messages = {
        "search": "🔍 Шукаю кандидатів...",
        "post": "📢 Розміщую вакансію в каналах...",
        "both": "🔍📢 Шукаю кандидатів та розміщую вакансію...",
        "skip": "⏸ Збережено. Вакансія чекає на обробку."
    }
    
    # Edit menu message to show status
    await edit_tg_message(chat_id, message_id, status_messages[action])
    
    # Load vacancy from DB
    vacancy = db.query(HuntVacancy).filter(HuntVacancy.id == vacancy_id).first()
    if not vacancy:
        return
    
    if action in ["search", "both"]:
        asyncio.create_task(
            run_hunt(vacancy_id, vacancy.raw_text, thread_id, chat_id)
        )
    
    if action in ["post", "both"]:
        asyncio.create_task(
            run_vacancy_posting(vacancy_id, chat_id, thread_id)
        )

from telethon import TelegramClient
from telethon.sessions import StringSession
import os, asyncio, logging
from datetime import datetime

logger = logging.getLogger(__name__)

async def run_vacancy_posting(vacancy_id: int, chat_id: int, thread_id: int):
    
    from db import SessionLocal
    from models import HuntVacancy, HuntSource
    from services.hunt_card_formatter import send_tg_message
    
    db = SessionLocal()
    
    try:
        vacancy = db.query(HuntVacancy).filter(
            HuntVacancy.id == vacancy_id
        ).first()
        if not vacancy:
            return
        
        # Format public post
        requirements = vacancy.requirements or ""
        req_lines = "\n".join([
            f"— {r.strip()}" 
            for r in requirements.split(",") if r.strip()
        ])
        salary_text = (
            f"До {vacancy.salary_max}$/міс" 
            if vacancy.salary_max 
            else "За домовленістю"
        )
        position_tag = (vacancy.position or "вакансія").replace(" ", "_").lower()
        city_tag = (vacancy.city or "україна").lower()
        hr_contact = os.getenv("HR_CONTACT", "@avtd_hr")
        
        formatted_post = f"""🔔 Вакансія | Торговий Дім АВ

💼 {vacancy.position or "Вакансія"}
📍 {vacancy.city or "Україна"}

📋 Вимоги:
{req_lines or "— За домовленістю"}

💰 {salary_text}

📩 Відгукнутись: {hr_contact}

#вакансія #{position_tag} #{city_tag} #AVTD"""

        # Load channels
        channels = db.query(HuntSource).filter(
            HuntSource.is_active == True
        ).all()
        channel_names = [c.tg_channel for c in channels]
        
        if not channel_names:
            channel_names = ["DniproVakansiiq", "kharkiv_robota1", 
                           "works_dnepr", "kieve_rabota"]
        
        # Telethon credentials
        session_string = (
            os.getenv("HR_TELETHON_SESSION") or 
            os.getenv("TELETHON_SESSION")
        )
        api_id = int(os.getenv("TELETHON_API_ID", "0"))
        api_hash = os.getenv("TELETHON_API_HASH", "")
        
        if not all([session_string, api_id, api_hash]):
            await send_tg_message(
                chat_id, thread_id,
                "⚠️ HR_TELETHON_SESSION не налаштовано.\n"
                "Додайте credentials в Render Environment Variables."
            )
            return
        
        # Post to channels
        posted = []
        failed = []
        
        async with TelegramClient(
            StringSession(session_string), api_id, api_hash
        ) as client:
            for channel in channel_names:
                try:
                    await client.send_message(channel, formatted_post)
                    posted.append(channel)
                    logger.info(f"Posted vacancy #{vacancy_id} to {channel}")
                    
                    # Save to hunt_postings
                    _save_posting(db, vacancy_id, channel, "posted")
                    
                    await asyncio.sleep(3)  # Rate limit
                    
                except Exception as e:
                    logger.error(f"Failed to post to {channel}: {e}")
                    failed.append(channel)
                    _save_posting(db, vacancy_id, channel, "failed", str(e))
        
        # Update vacancy status
        vacancy.status = "posted"
        db.commit()
        
        # Send report to supergroup thread
        report = f"📢 Вакансію розміщено!\n\n"
        report += f"✅ Успішно: {len(posted)} каналів\n"
        if failed:
            report += f"❌ Помилка: {len(failed)} каналів\n"
        report += f"\n📡 Охоплення: ~{len(posted) * 5000:,} підписників"
        report += f"\n\nКанали:\n"
        report += "\n".join([f"• @{c}" for c in posted])
        
        await send_tg_message(chat_id, thread_id, report)
        
    except Exception as e:
        logger.error(f"run_vacancy_posting error: {e}")
        await send_tg_message(
            chat_id, thread_id,
            f"❌ Помилка розміщення: {str(e)}"
        )
    finally:
        db.close()


def _save_posting(db, vacancy_id, channel, status, error=None):
    from models import HuntPosting
    posting = HuntPosting(
        vacancy_id=vacancy_id,
        channel=channel,
        status=status,
        error_message=error
    )
    db.add(posting)
    db.commit()

async def handle_hunt_hire(callback_query, candidate_id, db):
    from models import HuntCandidate, HuntVacancy
    from datetime import datetime
    
    candidate = db.query(HuntCandidate).filter(
        HuntCandidate.id == candidate_id
    ).first()
    
    if candidate:
        candidate.hr_decision = "hired"
        candidate.hired_at = datetime.now()
        
        # Mark vacancy as filled
        vacancy = db.query(HuntVacancy).filter(
            HuntVacancy.id == candidate.vacancy_id
        ).first()
        if vacancy:
            vacancy.status = "filled"
        
        db.commit()
        
        # Edit card message
        await edit_tg_message_reply_markup(
            callback_query["message"]["chat"]["id"],
            callback_query["message"]["message_id"],
            "🎯 Найнято через Maya Hunt!"
        )
        
        await answer_callback_query(
            callback_query["id"],
            "🎯 Кандидата відмічено як найнятого!"
        )

    posted_at TIMESTAMP DEFAULT NOW()
