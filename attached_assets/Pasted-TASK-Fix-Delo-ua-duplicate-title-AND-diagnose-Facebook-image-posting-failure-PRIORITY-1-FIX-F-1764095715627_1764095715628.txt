Update _clean_content to be more aggressive:
```python
def _clean_content(self, content: str, title: str) -> str:
    """Clean with better duplicate title removal"""
    if not content:
        return ""
    
    import re
    
    # Remove metadata
    content = re.sub(r'Новини компаній', '', content)
    
    # Get lines
    lines = [l.strip() for l in content.split('\n') if l.strip()]
    
    # Remove ALL occurrences of exact title
    # (not just first, remove all duplicates)
    filtered = [line for line in lines if line != title]
    
    # Add title back ONCE at the start
    filtered = [title] + filtered
    
    # Join everything
    text = ' '.join(filtered)
    text = re.sub(r'  +', ' ', text)
    
    # Add paragraph breaks
    text = re.sub(r'([.!?"]\s+)([A-ZА-ЯІЇЄ])', r'\1\n\n\2', text)
    text = re.sub(r'\n{3,}', '\n\n', text)
    
    return text.strip()
```
