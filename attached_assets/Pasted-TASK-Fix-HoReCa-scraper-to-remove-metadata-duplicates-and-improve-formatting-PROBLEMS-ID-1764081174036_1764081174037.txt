```python
def _clean_content(self, content: str, title: str) -> str:
    """Clean HoReCa article content"""
    if not content:
        return ""
    
    import re
    
    # Remove duplicate title at start
    if content.startswith(title):
        content = content[len(title):].strip()
    
    # Remove metadata and footer patterns
    patterns_to_remove = [
        r'Pro-HoReCa.*?\n',
        r'Статті.*?\n',
        r'Цікаве за цей тиждень:.*',
        r'Усі представлені фото.*',
        r'Новини Ресторанів.*',
        r'Аналітичні огляди.*',
        r'Тенденції ринку.*',
        r'Постачальники HoReCa.*',
    ]
    
    for pattern in patterns_to_remove:
        content = re.sub(pattern, '', content, flags=re.DOTALL)
    
    # Split into lines and clean
    lines = content.split('\n')
    cleaned_lines = []
    
    for line in lines:
        line = line.strip()
        # Skip empty, very short, or metadata lines
        if (line and 
            len(line) > 10 and 
            'Pro-HoReCa' not in line and
            'Цікаве за цей' not in line):
            cleaned_lines.append(line)
    
    # Join with double newline for paragraphs
    content = '\n\n'.join(cleaned_lines)
    
    # Fix excessive newlines
    content = re.sub(r'\n{3,}', '\n\n', content)
    
    return content.strip()
```
