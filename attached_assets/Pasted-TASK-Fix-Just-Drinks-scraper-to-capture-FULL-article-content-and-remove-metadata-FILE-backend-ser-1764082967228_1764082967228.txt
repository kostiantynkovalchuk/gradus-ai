```python
def _clean_content(self, content: str, title: str) -> str:
    """Clean Just Drinks article content"""
    if not content:
        return ""
    
    import re
    
    # Remove duplicate title at start
    if content.startswith(title + '\n' + title):
        content = content[len(title):].strip()
    elif content.startswith(title):
        # Check if next occurrence is also title
        lines = content.split('\n')
        if len(lines) > 1 and lines[1].strip() == title:
            content = '\n'.join(lines[2:]).strip()
    
    # Split into lines
    lines = content.split('\n')
    cleaned_lines = []
    
    for line in lines:
        line = line.strip()
        
        if not line:
            continue
        
        # Skip author names (2-3 words, proper case, no punctuation at end)
        words = line.split()
        if (len(words) <= 3 and 
            not line.endswith('.') and
            all(w[0].isupper() for w in words if w)):
            continue
        
        # Skip date patterns
        if re.match(r'\d{1,2}\s+\w+,?\s+\d{4}', line):
            continue
        
        cleaned_lines.append(line)
    
    content = '\n\n'.join(cleaned_lines)
    
    # Add paragraph breaks at sentence boundaries
    content = re.sub(r'\.(\s+)([A-ZА-ЯІЇЄ])', r'.\n\n\2', content)
    
    # Remove excessive newlines
    content = re.sub(r'\n{3,}', '\n\n', content)
    
    return content.strip()
```
