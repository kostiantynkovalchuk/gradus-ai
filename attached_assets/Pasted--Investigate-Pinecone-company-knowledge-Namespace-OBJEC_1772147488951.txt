## TASK 1: Query company_knowledge Namespace

Sample the content to see what's stored:
```python
import os
from pinecone import Pinecone
import json

# Initialize Pinecone
pc = Pinecone(api_key=os.getenv("PINECONE_API_KEY"))
index = pc.Index("gradus-media")

print("🔍 Investigating company_knowledge namespace...\n")

# Get index stats
stats = index.describe_index_stats()
print("📊 Index Stats:")
for ns_name, ns_stats in stats.get('namespaces', {}).items():
    print(f"  {ns_name}: {ns_stats.get('vector_count', 0)} vectors")

print("\n" + "="*60 + "\n")

# Query with dummy vector to get samples
print("📦 Sampling company_knowledge content:\n")

dummy_vector = [0.1] * 1536  # Dummy query vector

results = index.query(
    vector=dummy_vector,
    top_k=10,  # Get 10 samples
    namespace="company_knowledge",
    include_metadata=True
)

company_samples = []
for i, match in enumerate(results.matches, 1):
    sample = {
        'id': match.id,
        'score': match.score,
        'metadata': match.metadata
    }
    company_samples.append(sample)
    
    print(f"{i}. ID: {match.id}")
    print(f"   Metadata: {json.dumps(match.metadata, ensure_ascii=False, indent=2)}")
    print()

print("="*60 + "\n")
```

## TASK 4: Content Analysis

Analyze the differences:
```python
print("📊 Content Analysis:\n")

# Extract topics from metadata
def extract_topics(samples):
    topics = set()
    for sample in samples:
        metadata = sample.get('metadata', {})
        # Look for common metadata fields
        if 'title' in metadata:
            topics.add(metadata['title'][:50])
        if 'category' in metadata:
            topics.add(f"Category: {metadata['category']}")
        if 'source' in metadata:
            topics.add(f"Source: {metadata['source']}")
    return topics

company_topics = extract_topics(company_samples)
hr_topics = extract_topics(hr_samples)

print("company_knowledge topics:")
for topic in sorted(company_topics)[:15]:
    print(f"  • {topic}")

print("\nhr_docs topics:")
for topic in sorted(hr_topics)[:15]:
    print(f"  • {topic}")

print("\n" + "="*60 + "\n")

# Check for overlaps
overlap = company_topics & hr_topics
if overlap:
    print(f"⚠️ OVERLAP DETECTED ({len(overlap)} topics):")
    for topic in overlap:
        print(f"  • {topic}")
else:
    print("✅ No topic overlap - namespaces are distinct")

print("\n" + "="*60 + "\n")
```
