## WHAT YOU ARE BUILDING

A new Telegram bot (`@alex_avtd_bot`) for AVTD field sales agents.
It answers operational questions (KPI, Blitz Trade app, orders, routes, promotions)
and gives agents access to the full AVTD brand portfolio knowledge.

It uses the EXACT same RAG + preset infrastructure already serving Maya HR bot.
No new tables, no new embeddings, no new AI services.

Architecture:
- New bot token → new webhook route → same `HRRagService` + `hr_preset_answers` + Pinecone
- Auth: check `hr_users` table (same as Maya HR)
- Main menu: 5 InlineKeyboard buttons
- Photo Report: deep link button → navigates to existing @avtd_photo_report_bot
- System prompt: AVTD field agent framing (not HR framing)
- bot_source tag: `alex_avtd` (for analytics separation)

# ── Telegram API helpers ───────────────────────────────────────────────────

async def tg_send(chat_id: int, text: str, keyboard=None, parse_mode="Markdown"):
    payload = {"chat_id": chat_id, "text": text, "parse_mode": parse_mode}
    if keyboard:
        payload["reply_markup"] = keyboard
    async with httpx.AsyncClient() as client:
        await client.post(f"{TELEGRAM_API}/sendMessage", json=payload, timeout=10)

async def tg_answer_callback(callback_id: str):
    async with httpx.AsyncClient() as client:
        await client.post(f"{TELEGRAM_API}/answerCallbackQuery",
                         json={"callback_query_id": callback_id}, timeout=5)

async def tg_edit(chat_id: int, message_id: int, text: str, keyboard=None):
    payload = {"chat_id": chat_id, "message_id": message_id,
               "text": text, "parse_mode": "Markdown"}
    if keyboard:
        payload["reply_markup"] = keyboard
    async with httpx.AsyncClient() as client:
        await client.post(f"{TELEGRAM_API}/editMessageText", json=payload, timeout=10)

# ── Auth check ─────────────────────────────────────────────────────────────

def is_authorized(telegram_id: int, db: Session) -> bool:
    """Check hr_users table — same auth as Maya HR."""
    from sqlalchemy import text
    result = db.execute(
        text("SELECT 1 FROM hr_users WHERE telegram_id = :tid AND is_active = true"),
        {"tid": telegram_id}
    ).fetchone()
    return result is not None

# ── Log query to hr_query_log ──────────────────────────────────────────────

async def log_query(telegram_id: int, query: str, preset_matched: bool,
                    response_time_ms: int, db: Session):
    """Log agent query for analytics. bot_source = 'alex_avtd'."""
    try:
        from sqlalchemy import text
        db.execute(
            text("""
                INSERT INTO hr_query_log
                    (user_id, query, preset_matched, response_time_ms, bot_source, created_at)
                VALUES
                    (:uid, :q, :pm, :rt, 'alex_avtd', NOW())
            """),
            {"uid": telegram_id, "q": query[:500], "pm": preset_matched, "rt": response_time_ms}
        )
        db.commit()
    except Exception as e:
        logger.warning(f"[ALEX_AVTD] log_query failed: {e}")

# ── Main webhook ───────────────────────────────────────────────────────────

@alex_avtd_router.post("/api/telegram/alex_avtd_webhook")
async def alex_avtd_webhook(request: Request):
    try:
        update = await request.json()
        db: Session = next(get_db())

        # ── Callback query (button press) ──────────────────────────────
        if "callback_query" in update:
            cq = update["callback_query"]
            callback_id = cq["id"]
            telegram_id = cq["from"]["id"]
            chat_id = cq["message"]["chat"]["id"]
            message_id = cq["message"]["message_id"]
            data = cq.get("data", "")

            await tg_answer_callback(callback_id)

            # Auth check
            if not is_authorized(telegram_id, db):
                await tg_send(chat_id, "⛔ Доступ заборонено. Зверніться до HR.")
                return {"ok": True}

            # Main menu
            if data == "ax_menu":
                await tg_edit(chat_id, message_id,
                              "👋 Оберіть розділ:", main_menu_keyboard())
                return {"ok": True}

            # Sub-menus
            if data == "ax_client":
                await tg_edit(chat_id, message_id,
                              "👤 *Інформація по клієнту*\n\nОберіть тему:",
                              client_keyboard())
                return {"ok": True}
            if data == "ax_kpi":
                await tg_edit(chat_id, message_id,
                              "📊 *KPI та показники*\n\nОберіть тему:",
                              kpi_keyboard())
                return {"ok": True}
            if data == "ax_tech":
                await tg_edit(chat_id, message_id,
                              "🔧 *Технічні питання*\n\nОберіть тему:",
                              tech_keyboard())
                return {"ok": True}
            if data == "ax_orders":
                await tg_edit(chat_id, message_id,
                              "📦 *Замовлення та залишки*\n\nОберіть тему:",
                              orders_keyboard())
                return {"ok": True}

            # Preset answers
            if data in PRESETS:
                await tg_edit(chat_id, message_id, PRESETS[data], back_keyboard())
                await log_query(telegram_id, data, True, 50, db)
                return {"ok": True}

            return {"ok": True}

        # ── Regular message ────────────────────────────────────────────
        if "message" not in update:
            return {"ok": True}

        msg = update["message"]
        chat_id = msg["chat"]["id"]
        telegram_id = msg["from"]["id"]
        text = msg.get("text", "").strip()

        if not text:
            return {"ok": True}

        # Auth check
        if not is_authorized(telegram_id, db):
            await tg_send(
                chat_id,
                "⛔ *Доступ заборонено*\n\nЦей бот доступний лише для співробітників АВТД.\n"
                "Зверніться до HR-відділу для отримання доступу.",
            )
            return {"ok": True}

        # /start and /menu
        if text in ("/start", "/menu"):
            await tg_send(
                chat_id,
                "👋 Привіт! Я *Alex* — твій асистент у полі.\n\n"
                "Обери розділ або напиши питання напряму 👇",
                main_menu_keyboard()
            )
            return {"ok": True}

        # Free-text → RAG (same HRRagService as Maya HR)
        import time
        t0 = time.time()
        try:
            # Initialize HRRagService — import here to avoid circular imports
            # hr_pinecone_index is initialized in main app startup
            from backend.main import hr_pinecone_index
            service = HRRagService(
                pinecone_index=hr_pinecone_index,
                db_session=db,
                system_prompt_override=ALEX_AVTD_SYSTEM_PROMPT
            )
            answer = await service.get_answer_with_context(
                query=text,
                user_context={"user_id": telegram_id, "bot_source": "alex_avtd"}
            )
            response_text = answer.text
            preset_matched = answer.from_preset
        except Exception as e:
            logger.error(f"[ALEX_AVTD] RAG error: {e}")
            response_text = (
                "⚠️ Виникла помилка при обробці запиту.\n\n"
                "Спробуй ще раз або зверніться до:\n"
                "• Старшого оператора\n"
                "• Порталу: https://tdav.atlassian.net/"
            )
            preset_matched = False

        elapsed_ms = int((time.time() - t0) * 1000)
        await tg_send(chat_id, response_text, back_keyboard())
        await log_query(telegram_id, text, preset_matched, elapsed_ms, db)

    except Exception as e:
        logger.error(f"[ALEX_AVTD] Webhook error: {e}")

    return {"ok": True}
