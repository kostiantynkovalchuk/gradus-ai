```python
def _clean_content(self, content: str, title: str) -> str:
    """Clean Just Drinks article content"""
    if not content:
        return ""
    
    import re
    
    # Split into lines
    lines = content.split('\n')
    cleaned_lines = []
    
    for line in lines:
        line = line.strip()
        
        # Skip empty lines
        if not line:
            continue
        
        # Skip author names (2-3 words, all capitalized)
        words = line.split()
        if len(words) <= 3 and all(w[0].isupper() for w in words if w):
            continue
        
        # Skip date patterns
        if re.match(r'\d{1,2}\s+\w+,?\s+\d{4}', line):
            continue
        
        # Keep everything else (including title and source tag)
        cleaned_lines.append(line)
    
    # Join and add paragraph breaks
    content = '\n\n'.join(cleaned_lines)
    
    # Fix paragraph spacing at sentence boundaries
    content = re.sub(r'\.(\s+)([A-ZА-ЯІЇЄ])', r'.\n\n\2', content)
    
    # Remove excessive newlines
    content = re.sub(r'\n{3,}', '\n\n', content)
    
    return content.strip()
```
