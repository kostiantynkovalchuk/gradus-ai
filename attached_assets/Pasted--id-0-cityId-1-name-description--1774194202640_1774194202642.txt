async def close_vacancy_on_robotaua(robotaua_vacancy_id: int) -> bool:
    """Close a vacancy on Robota.ua when HR clicks 'Найняти' in Telegram."""
    token = await login_robotaua()
    async with httpx.AsyncClient(timeout=15) as client:
        resp = await client.post(
            f"https://employer-api.robota.ua/vacancy/state/{robotaua_vacancy_id}?state=Closed",
            headers={"Authorization": f"Bearer {token}"}
        )
        return resp.status_code == 200

async def get_avtd_vacancies() -> list[dict]:
    """List all AVTD vacancies on Robota.ua (for dashboard/sync)."""
    token = await login_robotaua()
    async with httpx.AsyncClient(timeout=15) as client:
        resp = await client.post(
            "https://employer-api.robota.ua/vacancy/list",
            headers={"Authorization": f"Bearer {token}", "Content-Type": "application/json"},
            json={"vacancyStateId": "4", "page": 0}
        )
        if resp.status_code == 200:
            return resp.json()
        return []
