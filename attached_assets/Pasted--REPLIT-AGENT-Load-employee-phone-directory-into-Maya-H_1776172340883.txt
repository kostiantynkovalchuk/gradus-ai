## STEP 1 — PARSE THE PHONE FILE

Write a one-time Python script (do NOT add to production code yet)
that reads `blitz_phones.xlsx` and produces clean structured data:

```python
import pandas as pd
import re

def parse_phone(phone_str):
    """Extract work and mobile numbers from blitz format."""
    if not phone_str:
        return None, None
    # Extract (р) work phone
    work = re.search(r'\(р\)\s*(\(\d{3}\)\d{3}-\d{2}-\d{2})', phone_str)
    # Extract (л) mobile phone  
    mobile = re.search(r'\(л\)\s*(\(\d{3}\)\d{3}-\d{2}-\d{2})', phone_str)
    work_num = work.group(1) if work else None
    mobile_num = mobile.group(1) if mobile else None
    return work_num, mobile_num

df = pd.read_excel('blitz_phones.xlsx')
employees = []
for _, row in df.iterrows():
    name = str(row['ФИО']).strip()
    work, mobile = parse_phone(str(row['Телехвон']))
    # Skip if both phones are empty
    if not work and not mobile:
        continue
    employees.append({
        'name': name,
        'work': work,
        'mobile': mobile
    })

print(f"Total employees with at least one phone: {len(employees)}")
print("Sample:", employees[:3])
```
