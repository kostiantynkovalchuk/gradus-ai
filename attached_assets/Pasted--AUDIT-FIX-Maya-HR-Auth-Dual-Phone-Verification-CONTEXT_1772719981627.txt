### 3a. Update `normalize_phone()` in phone_normalizer.py
```python
import re

def normalize_phone(phone: str) -> str | None:
    """
    Normalize any phone to 380XXXXXXXXX (12 digits).
    Returns None for invalid/non-normalizable numbers.
    
    Handles:
    - (095)810-00-72  → 380958100072
    - 067-631-33-99   → 380676313399  
    - +380956900289   → 380956900289
    - 0671234567      → 380671234567
    - 380XXXXXXXXX    → 380XXXXXXXXX (passthrough)
    - +34692480784    → None (non-UA, handled by whitelist)
    - (380)987-82-53  → None (only 10 digits after strip, bad data)
    """
    if not phone or not isinstance(phone, str):
        return None

    digits = re.sub(r'\D', '', phone)

    if len(digits) == 10 and digits.startswith('0'):
        return '380' + digits[1:]         # 0XXXXXXXXX → 380XXXXXXXXX
    elif len(digits) == 12 and digits.startswith('380'):
        return digits                      # already correct
    elif len(digits) == 13 and digits.startswith(('380', '0380')):
        return digits[-12:]                # strip leading extra digit
    else:
        return None                        # invalid — bad data or non-UA
```

### 3b. Update `verify_employee_in_sed()` in hr_sed_service.py

The goal: try the user's phone against SED, but ALSO check if
the returned employee has a second phone that matches.
```python
async def verify_employee_in_sed(user_phone: str) -> dict | None:
    """
    Verify employee by phone against SED API.
    
    Strategy:
    1. Normalize input phone
    2. Send to SED API
    3. If not found, try format variations (0XX vs 380XX)
    4. If found, also check if response contains alternate phone
    5. Return employee dict or None
    """
    normalized = normalize_phone(user_phone)
    if not normalized:
        logger.warning(f"Could not normalize phone: {user_phone}")
        return None

    # Generate format variations to try
    # SED may store as 0XXXXXXXXX or 380XXXXXXXXX — try both
    local_format = '0' + normalized[3:]   # 380XXXXXXXXX → 0XXXXXXXXX
    
    phones_to_try = list(dict.fromkeys([
        normalized,     # 380XXXXXXXXX
        local_format,   # 0XXXXXXXXX
        f'+{normalized}' # +380XXXXXXXXX
    ]))

    for phone_attempt in phones_to_try:
        logger.info(f"SED lookup attempt: {phone_attempt}")
        
        try:
            async with httpx.AsyncClient(timeout=10.0) as client:
                response = await client.post(
                    f"{SED_API_URL}/api/employees",
                    headers={"Authorization": f"Bearer {SED_API_KEY}",
                             "Content-Type": "application/json"},
                    json={"phone": phone_attempt}
                )

                if response.status_code != 200:
                    logger.warning(f"SED API non-200: {response.status_code}")
                    continue

                data = response.json()
                
                if data.get("status") and data.get("data"):
                    employee = data["data"]
                    logger.info(
                        f"✅ SED match: {phone_attempt} → "
                        f"{employee.get('full_name')}"
                    )
                    
                    # Log what phone fields SED returns (for audit)
                    logger.debug(f"SED employee fields: {list(employee.keys())}")
                    
                    return {
                        "verified": True,
                        "matched_phone": phone_attempt,
                        "employee": {
                            "employee_id": employee.get("employee_id"),
                            "full_name": employee.get("full_name"),
                            "first_name": employee.get("first_name"),
                            "last_name": employee.get("last_name"),
                            "department": employee.get("department"),
                            "position": employee.get("position"),
                            "phone": employee.get("phone"),
                            "phone_work": employee.get("phone_work"),   # (р) if SED returns it
                            "phone_mobile": employee.get("phone_mobile"), # (л) if SED returns it
                            "start_date": employee.get("start_date"),
                            "email": employee.get("email")
                        }
                    }

        except httpx.TimeoutException:
            logger.error(f"SED API timeout for {phone_attempt}")
        except Exception as e:
            logger.error(f"SED API error for {phone_attempt}: {e}")

    logger.warning(f"❌ No SED match found for: {user_phone}")
    return None
```
