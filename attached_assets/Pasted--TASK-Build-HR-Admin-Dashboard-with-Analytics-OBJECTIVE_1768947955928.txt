```python
"""
Helper script to extract CSS variables and styling from main site
Run this first to ensure consistent design
"""

import httpx
import re
from bs4 import BeautifulSoup

async def inspect_main_site():
    """Extract styling information from main gradus-ai site"""
    
    async with httpx.AsyncClient() as client:
        try:
            response = await client.get('https://gradus-ai.onrender.com/')
            
            if response.status_code == 200:
                soup = BeautifulSoup(response.text, 'html.parser')
                
                # Extract CSS
                styles = soup.find_all('style')
                for style in styles:
                    print("=== FOUND STYLE BLOCK ===")
                    print(style.string[:500] if style.string else "Empty")
                    print()
                
                # Extract color scheme
                print("=== EXTRACTING COLORS ===")
                style_text = ' '.join([s.string for s in styles if s.string])
                
                # Common CSS color patterns
                colors = re.findall(r'(?:color|background|border):\s*(#[0-9a-fA-F]{3,6}|rgb\([^)]+\)|rgba\([^)]+\))', style_text)
                unique_colors = list(set(colors))
                
                print("Found colors:", unique_colors[:20])
                
                # Extract fonts
                fonts = re.findall(r'font-family:\s*([^;]+)', style_text)
                print("\nFonts:", list(set(fonts))[:5])
                
        except Exception as e:
            print(f"Error: {e}")
            print("\nUsing fallback design system...")

if __name__ == "__main__":
    import asyncio
    asyncio.run(inspect_main_site())
```

```python
def verify_admin(credentials: HTTPBasicCredentials = Depends(security)):
    """Verify HTTP Basic Auth credentials"""
    correct_username = secrets.compare_digest(credentials.username, ADMIN_USERNAME)
    correct_password = secrets.compare_digest(credentials.password, ADMIN_PASSWORD)

    return credentials.username

@router.get("/api/overview")
async def get_overview(
    days: int = 7,
    credentials: HTTPBasicCredentials = Depends(verify_admin)
):
    """Get overview statistics"""
    
    end_date = date.today()
    start_date = end_date - timedelta(days=days)
    
    async with get_db() as db:
        # Main stats
        result = await db.execute(text("""
            SELECT
                COUNT(*) as total_queries,
                COUNT(*) FILTER (WHERE preset_matched = TRUE) as preset_hits,
                COUNT(*) FILTER (WHERE rag_used = TRUE) as rag_queries,
                COUNT(DISTINCT user_id) as unique_users,
                AVG(response_time_ms)::INTEGER as avg_response_time,
                (COUNT(*) FILTER (WHERE satisfied = TRUE)::DECIMAL / 
                 NULLIF(COUNT(*) FILTER (WHERE satisfied IS NOT NULL), 0) * 100)::DECIMAL(5,2) as satisfaction_rate
            FROM hr_query_log
            WHERE DATE(created_at) >= :start_date
        """), {'start_date': start_date})
        
        stats = result.fetchone()
        
        total = stats[0] or 0
        preset_rate = (stats[1] / total * 100) if total > 0 else 0
        rag_rate = (stats[2] / total * 100) if total > 0 else 0
    
    return {
        'period': {
            'start_date': str(start_date),
            'end_date': str(end_date),
            'days': days
        },
        'stats': {
            'total_queries': total,
            'preset_hits': stats[1] or 0,
            'preset_coverage': round(preset_rate, 1),
            'rag_queries': stats[2] or 0,
            'rag_usage': round(rag_rate, 1),
            'unique_users': stats[3] or 0,
            'avg_response_time_ms': stats[4] or 0,
            'satisfaction_rate': float(stats[5]) if stats[5] else 0.0
        }
    }

@router.get("/api/top-queries")
async def get_top_queries(
    days: int = 7,
    limit: int = 20,
    credentials: HTTPBasicCredentials = Depends(verify_admin)
):
    """Get most popular queries"""
    
    start_date = date.today() - timedelta(days=days)
    
    async with get_db() as db:
        result = await db.execute(text("""
            SELECT 
                query_normalized,
                COUNT(*) as frequency,
                COUNT(*) FILTER (WHERE preset_matched = TRUE) as preset_hits,
                COUNT(*) FILTER (WHERE satisfied = TRUE) as helpful_count,
                COUNT(*) FILTER (WHERE satisfied = FALSE) as not_helpful_count,
                COUNT(*) FILTER (WHERE satisfied IS NOT NULL) as feedback_count,
                AVG(response_time_ms)::INTEGER as avg_response_time
            FROM hr_query_log
            WHERE DATE(created_at) >= :start_date
            GROUP BY query_normalized
            ORDER BY frequency DESC
            LIMIT :limit
        """), {'start_date': start_date, 'limit': limit})
        
        queries = result.fetchall()
    
    return {
        'queries': [
            {
                'query': q[0],
                'frequency': q[1],
                'preset_hits': q[2],
                'helpful_count': q[3],
                'not_helpful_count': q[4],
                'feedback_count': q[5],
                'satisfaction_rate': round((q[3] / q[5] * 100) if q[5] > 0 else 0, 1),
                'avg_response_time_ms': q[6],
                'has_feedback': q[5] > 0
            }
            for q in queries
        ]
    }

@router.get("/api/satisfaction-breakdown")
async def get_satisfaction_breakdown(
    days: int = 7,
    credentials: HTTPBasicCredentials = Depends(verify_admin)
):
    """Get satisfaction breakdown by category"""
    
    start_date = date.today() - timedelta(days=days)
    
    async with get_db() as db:
        # Get top dissatisfied queries
        result = await db.execute(text("""
            SELECT 
                query,
                COUNT(*) as frequency,
                COUNT(*) FILTER (WHERE satisfied = FALSE) as not_helpful,
                array_agg(DISTINCT user_name) FILTER (WHERE user_name IS NOT NULL) as users
            FROM hr_query_log
            WHERE 
                DATE(created_at) >= :start_date
                AND satisfied = FALSE
            GROUP BY query
            ORDER BY frequency DESC
            LIMIT 10
        """), {'start_date': start_date})
        
        dissatisfied = result.fetchall()
        
        # Get top satisfied queries
        result2 = await db.execute(text("""
            SELECT 
                query,
                COUNT(*) as frequency,
                COUNT(*) FILTER (WHERE satisfied = TRUE) as helpful
            FROM hr_query_log
            WHERE 
                DATE(created_at) >= :start_date
                AND satisfied = TRUE
            GROUP BY query
            ORDER BY frequency DESC
            LIMIT 10
        """), {'start_date': start_date})
        
        satisfied = result2.fetchall()
    
    return {
        'not_helpful': [
            {
                'query': q[0],
                'frequency': q[1],
                'not_helpful_count': q[2],
                'affected_users': q[3][:3] if q[3] else []
            }
            for q in dissatisfied
        ],
        'helpful': [
            {
                'query': q[0],
                'frequency': q[1],
                'helpful_count': q[2]
            }
            for q in satisfied
        ]
    }

@router.get("/api/preset-candidates")
async def get_preset_candidates(
    min_frequency: int = 3,
    days: int = 30,
    credentials: HTTPBasicCredentials = Depends(verify_admin)
):
    """Find queries that should become presets"""
    
    start_date = date.today() - timedelta(days=days)
    
    async with get_db() as db:
        result = await db.execute(text("""
            SELECT 
                query_normalized,
                COUNT(*) as frequency,
                AVG(response_time_ms)::INTEGER as avg_time,
                COUNT(*) FILTER (WHERE satisfied = FALSE) as dissatisfied_count,
                COUNT(*) FILTER (WHERE satisfied IS NOT NULL) as feedback_count
            FROM hr_query_log
            WHERE 
                preset_matched = FALSE
                AND DATE(created_at) >= :start_date
            GROUP BY query_normalized
            HAVING COUNT(*) >= :min_freq
            ORDER BY frequency DESC, dissatisfied_count DESC
            LIMIT 20
        """), {'start_date': start_date, 'min_freq': min_frequency})
        
        candidates = result.fetchall()
    
    return {
        'candidates': [
            {
                'query': c[0],
                'frequency': c[1],
                'avg_response_time_ms': c[2],
                'dissatisfied_count': c[3],
                'feedback_count': c[4],
                'priority': 'high' if c[1] >= 10 else ('medium' if c[1] >= 5 else 'low'),
                'should_add': c[1] >= 5 or c[3] >= 2
            }
            for c in candidates
        ]
    }

@router.post("/api/presets")
async def create_preset(
    pattern: str,
    answer: str,
    content_ids: list[str],
    priority: int = 5,
    credentials: HTTPBasicCredentials = Depends(verify_admin)
):
    """Create new preset answer"""
    
    async with get_db() as db:
        content_ids_array = '{' + ','.join(f'"{c}"' for c in content_ids) + '}'
        
        result = await db.execute(text("""
            INSERT INTO hr_preset_answers (
                question_pattern,
                answer_text,
                content_ids,
                priority,
                is_active
            ) VALUES (
                :pattern,
                :answer,
                :content_ids,
                :priority,
                true
            )
            RETURNING id
        """), {
            'pattern': pattern,
            'answer': answer,
            'content_ids': content_ids_array,
            'priority': priority
        })
        
        preset_id = result.scalar()
        await db.commit()
    
    return {
        'status': 'success',
        'preset_id': preset_id,
        'pattern': pattern
    }

@router.patch("/api/presets/{preset_id}/toggle")
async def toggle_preset(
    preset_id: int,
    credentials: HTTPBasicCredentials = Depends(verify_admin)
):
    """Toggle preset active status"""
    
    async with get_db() as db:
        result = await db.execute(text("""
            UPDATE hr_preset_answers
            SET is_active = NOT is_active
            WHERE id = :id
            RETURNING is_active
        """), {'id': preset_id})
        
        new_status = result.scalar()
        await db.commit()
    
    return {
        'status': 'success',
        'preset_id': preset_id,
        'is_active': new_status
    }
```

```html
        let currentPeriod = 7;

        document.addEventListener('DOMContentLoaded', () => {
            loadDashboard();

        async function loadDashboard() {
            await Promise.all([
                loadOverview(),
                loadTopQueries(),
                loadSatisfactionBreakdown(),
                loadPresetCandidates(),
                loadRecentActivity()
            ]);
        }

        async function loadSystemInfo() {
            try {
                const response = await fetch('/hr/api/system-info');
                const data = await response.json();
                document.getElementById('systemInfo').textContent = 
                    `${data.content_items} content items ‚Ä¢ ${data.active_presets} active presets ‚Ä¢ ${data.embeddings} embeddings`;
            } catch (error) {
                console.error('Error loading system info:', error);
            }
        }

        async function loadOverview() {
            try {
                const response = await fetch(`/hr/api/overview?days=${currentPeriod}`);
                const data = await response.json();
                const stats = data.stats;

                document.getElementById('totalQueries').textContent = stats.total_queries.toLocaleString('uk-UA');
                document.getElementById('presetCoverage').textContent = stats.preset_coverage.toFixed(1) + '%';
                document.getElementById('satisfaction').textContent = stats.satisfaction_rate.toFixed(1) + '%';
                document.getElementById('uniqueUsers').textContent = stats.unique_users.toLocaleString('uk-UA');

                document.getElementById('presetProgress').style.width = stats.preset_coverage + '%';
                document.getElementById('satisfactionProgress').style.width = stats.satisfaction_rate + '%';
                
                const satisfactionBar = document.getElementById('satisfactionProgress');
                satisfactionBar.className = 'progress-fill ' + 
                    (stats.satisfaction_rate >= 80 ? 'success' : 
                     stats.satisfaction_rate >= 60 ? 'warning' : 'danger');

            } catch (error) {
                console.error('Error loading overview:', error);
            }
        }

        async function loadTopQueries() {
            const loading = document.getElementById('topQueriesLoading');
            const table = document.getElementById('topQueriesTable');
            const tbody = document.getElementById('topQueriesBody');

            try {
                const response = await fetch(`/hr/api/top-queries?days=${currentPeriod}&limit=20`);
                const data = await response.json();

                tbody.innerHTML = '';

                if (data.queries.length === 0) {
                    loading.innerHTML = '<div class="empty-state"><div class="empty-state-icon">üì≠</div><p class="empty-state-text">–ù–µ–º–∞—î –¥–∞–Ω–∏—Ö –∑–∞ –æ–±—Ä–∞–Ω–∏–π –ø–µ—Ä—ñ–æ–¥</p></div>';
                    return;
                }

                data.queries.forEach((query, index) => {
                    const row = document.createElement('tr');
                    
                    const satisfactionClass = 
                        query.satisfaction_rate >= 80 ? 'success' :
                        query.satisfaction_rate >= 60 ? 'warning' : 'danger';

                    row.innerHTML = `
                        <td>
                            <strong>${index + 1}.</strong> ${query.query}
                            ${query.preset_hits > 0 ? '<span class="badge badge-primary">‚ö° –ü—Ä–µ—Å–µ—Ç</span>' : ''}
                        </td>
                        <td style="text-align: center;"><strong>${query.frequency}</strong></td>
                        <td style="text-align: center;">${query.preset_hits > 0 ? '‚úÖ' : '‚ùå'}</td>
                        <td style="text-align: center;">${query.helpful_count}</td>
                        <td style="text-align: center;">${query.not_helpful_count}</td>
                        <td style="text-align: center;">
                            ${query.has_feedback ? 
                                `<span class="badge badge-${satisfactionClass}">${query.satisfaction_rate.toFixed(0)}%</span>` : 
                                '<span style="color: var(--gray-400);">-</span>'}
                        </td>
                    `;
                    tbody.appendChild(row);
                });

                loading.style.display = 'none';
                table.style.display = 'table';

            } catch (error) {
                console.error('Error loading top queries:', error);
                loading.innerHTML = '<div class="empty-state"><p class="empty-state-text">–ü–æ–º–∏–ª–∫–∞ –∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–Ω—è –¥–∞–Ω–∏—Ö</p></div>';
            }
        }

                        <div class="query-item">

        async function loadPresetCandidates() {

            const list = document.getElementById('candidatesList');

            try {
                const response = await fetch(`/hr/api/preset-candidates?days=${currentPeriod}`);

                    loading.style.display = 'none';
                    return;
                }

                list.innerHTML = data.candidates.map(c => {
                    const priorityBadge = 
                        c.priority === 'high' ? 'badge-danger' :
                        c.priority === 'medium' ? 'badge-warning' : 'badge-primary';

                    return `
                        <div class="query-item">
                            <div class="query-text">
                                üí° ${c.query}
                                ${c.should_add ? '<span class="badge badge-warning">–†–µ–∫–æ–º–µ–Ω–¥–æ–≤–∞–Ω–æ</span>' : ''}
                            </div>
                            <div class="query-meta">
                                <span class="badge ${priorityBadge}">${c.priority}</span>
                                <span>üî¢ ${c.frequency}x</span>
                                ${c.dissatisfied_count > 0 ? `<span>üòû ${c.dissatisfied_count}</span>` : ''}
                                <span>‚è±Ô∏è ${c.avg_response_time_ms}ms</span>
                            </div>
                        </div>
                    `;
                }).join('');

                loading.style.display = 'none';

            } catch (error) {
```
