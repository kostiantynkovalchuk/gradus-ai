-- Table 1: survey definitions
CREATE TABLE IF NOT EXISTS hr_surveys (
    id          SERIAL PRIMARY KEY,
    survey_id   VARCHAR(64) UNIQUE NOT NULL,
    question    TEXT NOT NULL,
    options     JSONB NOT NULL,
    is_open     BOOLEAN DEFAULT TRUE,
    sent_count  INTEGER DEFAULT 0,
    created_at  TIMESTAMP DEFAULT NOW(),
    opens_at    TIMESTAMP
);

-- Table 2: individual votes
CREATE TABLE IF NOT EXISTS hr_survey_votes (
    id          SERIAL PRIMARY KEY,
    survey_id   VARCHAR(64) NOT NULL
                REFERENCES hr_surveys(survey_id) ON DELETE CASCADE,
    user_id     INTEGER NOT NULL REFERENCES hr_users(id),
    answer      VARCHAR(16) NOT NULL,
    voted_at    TIMESTAMP DEFAULT NOW(),
    UNIQUE (survey_id, user_id)
);

-- Table 3: scoreboard message tracking
CREATE TABLE IF NOT EXISTS hr_survey_meta (
    survey_id             VARCHAR(64) PRIMARY KEY
                          REFERENCES hr_surveys(survey_id),
    scoreboard_targets    JSONB,
    -- format: [{"chat_id": 123456, "msg_id": 789}, ...]
    -- one entry per observer (Konstantin + Natalia)
    last_edit_at          TIMESTAMP,
    final_result          VARCHAR(16),
    closed_at             TIMESTAMP
);

async def broadcast_survey(survey_id: str) -> dict:
  """
  Sends survey to all active hr_users with inline keyboard.
  Mirrors pattern from broadcast_avpost.py exactly.
  """
  1. Fetch survey from hr_surveys by survey_id
  2. Fetch all hr_users where telegram_id IS NOT NULL AND is_active = true
  3. Build Telegram payload:
       {
         "chat_id": <telegram_id>,
         "text": <survey question>,
         "parse_mode": "Markdown",
         "reply_markup": {
           "inline_keyboard": [[
             {"text": "✅ Так", 
              "callback_data": "survey_easter_holiday_2026_yes"},
             {"text": "❌ Ні",  
              "callback_data": "survey_easter_holiday_2026_no"}
           ]]
         }
       }
  4. Send to each user via:
       POST https://api.telegram.org/bot{TOKEN}/sendMessage
     With asyncio.sleep(0.05) between sends
  5. Wrap each send in try/except — never crash the loop
  6. UPDATE hr_surveys SET sent_count = <successful sends>
     WHERE survey_id = :survey_id
  7. Return {"sent": N, "failed": M}

FUNCTION 3: get_results()

async def get_results(survey_id: str) -> dict:
  """Returns current vote counts and percentages."""
  
  Query:
    SELECT answer, COUNT(*) as cnt
    FROM hr_survey_votes
    WHERE survey_id = :survey_id
    GROUP BY answer
  
  Also fetch sent_count from hr_surveys.
  
  Build result:
    total = yes_count + no_count
    yes_pct = round(yes_count / total * 100, 1) if total > 0 else 0.0
    no_pct  = round(no_count  / total * 100, 1) if total > 0 else 0.0
  
  Return:
    {
      "yes":   {"count": N, "pct": 68.5},
      "no":    {"count": M, "pct": 31.5},
      "total": N + M,
      "sent":  hr_surveys.sent_count
    }

FUNCTION 4: _build_scoreboard_text()
─────────────────────────────────────
async def _build_scoreboard_text(survey_id: str) -> str:
  """Shared text builder for post and update."""
  
  r = await get_results(survey_id)
  kyiv_time = datetime.now(ZoneInfo("Europe/Kiev")).strftime("%H:%M")
  
  def bar(pct):
    filled = int(pct / 10)
    return "█" * filled + "░" * (10 - filled)
  
  return (
    f"📊 *Опитування: Вихідний 13/04 — LIVE*\n\n"
    f"✅ Так {bar(r['yes']['pct'])} "
        f"{r['yes']['pct']}% ({r['yes']['count']})\n"
    f"❌ Ні  {bar(r['no']['pct'])}  "
        f"{r['no']['pct']}% ({r['no']['count']})\n\n"
    f"👥 Проголосувало: {r['total']} з {r['sent']}\n"
    f"🕐 Оновлено: {kyiv_time}"
  )

FUNCTION 5: post_scoreboard()

async def post_scoreboard(survey_id: str):
  """
  Posts initial scoreboard to Konstantin + Natalia personal chats.
  Called once at 07:05 UTC by scheduler.
  Stores message_ids for future edits.
  """
  text = await _build_scoreboard_text(survey_id)
  observers = await get_survey_observers()
  message_ids = []

  async with httpx.AsyncClient() as client:
    for tg_id in observers:
      try:
        resp = await client.post(
          f"https://api.telegram.org/bot{TOKEN}/sendMessage",
          json={
            "chat_id": tg_id,
            "text": text,
            "parse_mode": "Markdown"
          }
        )
        msg_id = resp.json()["result"]["message_id"]
        message_ids.append({"chat_id": tg_id, "msg_id": msg_id})
      except Exception as e:
        logger.error(f"post_scoreboard failed for {tg_id}: {e}")

  # Store targets JSON in hr_survey_meta
  INSERT INTO hr_survey_meta (survey_id, scoreboard_targets)
  VALUES (:survey_id, :targets)
  ON CONFLICT (survey_id) DO UPDATE
  SET scoreboard_targets = EXCLUDED.scoreboard_targets

FUNCTION 6: update_scoreboard()
─────────────────────────────────────
async def update_scoreboard(survey_id: str):
  """
  Edits existing scoreboard messages in Konstantin + Natalia chats.
  Called on every vote from callback handler.
  Has 3-second debounce to respect Telegram rate limits.
  """
  1. Fetch hr_survey_meta for survey_id
  2. If no record or no scoreboard_targets → return silently
  3. Debounce check:
       if last_edit_at is not None:
         elapsed = (datetime.utcnow() - last_edit_at).total_seconds()
         if elapsed < 3:
           return
  4. text = await _build_scoreboard_text(survey_id)
  5. targets = json.loads(scoreboard_targets)
  6. async with httpx.AsyncClient() as client:
       for target in targets:
         try:
           await client.post(
             f"https://api.telegram.org/bot{TOKEN}/editMessageText",
             json={
               "chat_id": target["chat_id"],
               "message_id": target["msg_id"],
               "text": text,
               "parse_mode": "Markdown"
             }
           )
         except Exception as e:
           if "message is not modified" not in str(e).lower():
             logger.error(f"update_scoreboard failed: {e}")
  7. UPDATE hr_survey_meta SET last_edit_at = NOW()
     WHERE survey_id = :survey_id

async def register_vote(
    survey_id: str,
    telegram_id: int,
    answer: str
) -> dict:
  """Records or updates a user's vote."""
  
  1. Fetch survey — check is_open
     If not open: return {"status": "closed"}
  
  2. Fetch hr_user by telegram_id
     If not found: return {"status": "unauthorized"}
  
  3. INSERT INTO hr_survey_votes (survey_id, user_id, answer)
     VALUES (:survey_id, :user_id, :answer)
     ON CONFLICT (survey_id, user_id)
     DO UPDATE SET answer = EXCLUDED.answer, voted_at = NOW()
  
  4. results = await get_results(survey_id)
  
  5. Return:
     {"status": "ok", "answer": answer, "results": results}

async def close_survey(survey_id: str):
  """
  Manually closes survey. Called only from HR dashboard button.
  Never called automatically by scheduler.
  """
  1. UPDATE hr_surveys SET is_open = FALSE WHERE survey_id = :survey_id
  
  2. results = await get_results(survey_id)
  
  3. Determine winner:
     winner = "yes" if yes_count >= no_count else "no"
     winner_label = "✅ Так" if winner == "yes" else "❌ Ні"
  
  4. Build final message:
       f"📊 *Опитування завершено: Вихідний 13/04*\n\n"
       f"✅ Так {bar} {pct}% ({count})\n"
       f"❌ Ні  {bar} {pct}% ({count})\n\n"
       f"👥 Всього проголосувало: {total} з {sent}\n"
       f"🏆 Результат: *{winner_label}*"
  
  5. Edit scoreboard messages in both personal chats with final text
     (same editMessageText pattern as update_scoreboard)
  
  6. UPDATE hr_survey_meta
     SET final_result = :winner, closed_at = NOW()
     WHERE survey_id = :survey_id

async function loadSurveyResults() {

    console.error("Survey load error:", err);

async function closeSurvey() {
  if (!confirm(
    "Закрити опитування?\nГолосування буде зупинено. Дію не можна скасувати."
  )) return;

  try {
    const res = await fetch(
      `/api/hr/survey/${SURVEY_ID}/close`,
      { method: "POST" }
    );
    if (res.ok) {
      loadSurveyResults();   // immediate refresh
    }
  } catch (err) {
    console.error("Close survey error:", err);
  }
}
