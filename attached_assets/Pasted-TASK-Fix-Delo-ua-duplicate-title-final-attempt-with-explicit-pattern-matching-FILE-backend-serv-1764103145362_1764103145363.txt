def _clean_content(self, content: str, title: str) -> str:
    """Clean Delo.ua content - explicit duplicate title removal"""
    if not content:
        return ""
    
    import re
    
    # Remove "Новини компаній" prefix
    content = re.sub(r'Новини компаній', '', content)
    
    # EXPLICIT FIX: Check if content starts with title repeated
    # Pattern: "Title\nTitle Rest of content"
    title_pattern = re.escape(title)
    
    # Check if title appears twice at the very start
    double_title_pattern = f'^{title_pattern}\\s*\\n\\s*{title_pattern}'
    if re.match(double_title_pattern, content, re.MULTILINE):
        # Remove first occurrence only
        content = re.sub(f'^{title_pattern}\\s*\\n', '', content, count=1, flags=re.MULTILINE)
    
    # Also check if title appears twice on same line with space between
    double_title_same_line = f'{title_pattern}\\s+{title_pattern}'
    content = re.sub(double_title_same_line, title, content)
    
    # Get lines and clean
    lines = [l.strip() for l in content.split('\n') if l.strip()]
    
    # Skip metadata lines
    filtered = []
    for line in lines:
        if line in ['Категорія', 'Новини', 'Дата публікації']:
            continue
        filtered.append(line)
    
    # Join all text
    text = ' '.join(filtered)
    
    # Fix spacing
    text = re.sub(r'  +', ' ', text)
    
    # Add paragraph breaks at sentence boundaries
    text = re.sub(r'([.!?»"]\s+)([A-ZА-ЯІЇЄ])', r'\1\n\n\2', text)
    
    # Clean excessive newlines
    text = re.sub(r'\n{3,}', '\n\n', text)
    
    return text.strip()
