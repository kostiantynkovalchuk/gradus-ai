import os, time, logging, httpx

logger = logging.getLogger(__name__)

_token_cache = {"token": None, "expires_at": 0}
_AUTH_URL = "https://auth-api.robota.ua"
_TOKEN_TTL = 23 * 3600  # refresh every 23 hours

async def login_robotaua() -> str | None:
    """Get valid Bearer token. Auto-login if expired."""
    now = time.time()
    if _token_cache["token"] and now < _token_cache["expires_at"]:
        return _token_cache["token"]
    
    email = os.getenv("ROBOTAUA_EMAIL")
    password = os.getenv("ROBOTAUA_PASSWORD")
    
    if not email or not password:
        # Fallback to manual JWT
        jwt = os.getenv("ROBOTAUA_JWT")
        if jwt:
            logger.warning("ROBOTAUA_EMAIL/PASSWORD not set, using ROBOTAUA_JWT fallback")
            return jwt
        logger.error("No Robota.ua credentials available")
        return None
    
    try:
        async with httpx.AsyncClient(timeout=15) as client:
            resp = await client.post(f"{_AUTH_URL}/Login", json={
                "username": email,
                "password": password,
                "remember": True
            })
            if resp.status_code == 200:
                token = resp.text.strip().strip('"')
                _token_cache["token"] = token
                _token_cache["expires_at"] = now + _TOKEN_TTL
                logger.info("Robota.ua login successful, token cached for 23h")
                return token
            else:
                logger.error(f"Robota.ua login failed: {resp.status_code} {resp.text[:200]}")
                # Fallback to env JWT
                jwt = os.getenv("ROBOTAUA_JWT")
                if jwt:
                    return jwt
                return None
    except Exception as e:
        logger.error(f"Robota.ua login error: {e}")
        jwt = os.getenv("ROBOTAUA_JWT")
        return jwt

def invalidate_token():
    """Call this on 401 response to force re-login on next call."""
    _token_cache["token"] = None
    _token_cache["expires_at"] = 0

async def get_city_id(city_name: str) -> int | None:
    """Resolve Ukrainian city name to Robota.ua city ID."""
    if not _city_cache:
        await _load_city_cache()
    
    name = city_name.strip().lower()
    if name in _city_cache:

    # Try partial match
    for cached_name, city_id in _city_cache.items():
        if name in cached_name or cached_name in name:
            return city_id
    return None

async def _load_city_cache():
    token = await login_robotaua()
    if not token:
        return
    try:
        async with httpx.AsyncClient(timeout=15) as client:
            resp = await client.get(
                f"{_EMPLOYER_API}/values/citylist",
                headers={"Authorization": f"Bearer {token}"}
            )
            if resp.status_code == 200:
                cities = resp.json()
                for city in cities:
                    name = city.get("name", "").strip().lower()
                    city_id = city.get("id")

                        _city_cache[name] = city_id
                        # Add common aliases
                        _add_city_aliases(name, city_id)
                logger.info(f"Loaded {len(_city_cache)} city mappings from Robota.ua")

        logger.error(f"Failed to load city cache: {e}")
