# === RISK SCORE ENGINE ===

RISK_POINTS = {
    "resignation": 3,
    "conflict": 1,
    "rights": 1,
    "burnout": 1,
}

RISK_THRESHOLD_ALERT = 4
RISK_THRESHOLD_URGENT = 7

def update_risk_score(employee_id: int, employee_name: str, department: str, points: int):
    """
    Update rolling risk score for an employee.
    Uses UPSERT — creates if first signal, increments if exists.
    """
    db = SessionLocal()
    try:
        db.execute(text("""
            INSERT INTO pulse_risk_scores 
                (employee_id, employee_name, department, current_score, last_trigger_at, last_calculated_at)
            VALUES (:eid, :ename, :dept, :pts, NOW(), NOW())
            ON CONFLICT (employee_id) DO UPDATE SET
                current_score = pulse_risk_scores.current_score + :pts,
                employee_name = :ename,
                department = :dept,
                last_trigger_at = NOW(),
                last_calculated_at = NOW(),
                alert_status = CASE
                    WHEN pulse_risk_scores.current_score + :pts >= :urgent THEN 'urgent'
                    WHEN pulse_risk_scores.current_score + :pts >= :alert THEN 'red'
                    ELSE pulse_risk_scores.alert_status
                END
        """), {
            "eid": employee_id, "ename": employee_name, "dept": department,
            "pts": points, "urgent": RISK_THRESHOLD_URGENT, "alert": RISK_THRESHOLD_ALERT
        })
        db.commit()
        
        # Check if threshold crossed → return True if HR alert needed
        result = db.execute(text(
            "SELECT current_score FROM pulse_risk_scores WHERE employee_id = :eid"
        ), {"eid": employee_id}).fetchone()
        
        return result[0] >= RISK_THRESHOLD_ALERT if result else False
    except Exception as e:
        logger.error(f"Risk score update failed: {e}")
        db.rollback()
        return False
    finally:
        db.close()


def get_risk_history(employee_id: int) -> list:
    """Get last 5 triggers for this employee within 30 days"""
    db = SessionLocal()
    try:
        rows = db.execute(text("""
            SELECT trigger_type, severity, risk_points, fired_at
            FROM pulse_triggers
            WHERE employee_id = :eid AND fired_at > NOW() - INTERVAL '30 days'
            ORDER BY fired_at DESC LIMIT 5
        """), {"eid": employee_id}).fetchall()
        return rows
    finally:
        db.close()


def log_hr_action(trigger_id: int, action: str, hr_user: str):
    """Log HR response to an alert"""
    db = SessionLocal()
    try:
        # Get employee_id from trigger
        row = db.execute(text(
            "SELECT employee_id FROM pulse_triggers WHERE id = :tid"
        ), {"tid": trigger_id}).fetchone()
        
        risk_score_id = None
        if row:
            r = db.execute(text(
                "SELECT id FROM pulse_risk_scores WHERE employee_id = :eid"
            ), {"eid": row[0]}).fetchone()
            risk_score_id = r[0] if r else None
        
        db.execute(text("""
            INSERT INTO pulse_hr_actions (trigger_id, risk_score_id, action, hr_user)
            VALUES (:tid, :rsid, :action, :hr)
        """), {"tid": trigger_id, "rsid": risk_score_id, "action": action, "hr": hr_user})
        
        # If resolved/false_positive → reduce risk score by 2
        if action in ("resolved", "false_positive") and row:
            db.execute(text("""
                UPDATE pulse_risk_scores 
                SET current_score = GREATEST(0, current_score - 2),
                    alert_status = CASE WHEN current_score - 2 <= 0 THEN 'none' ELSE alert_status END,
                    last_calculated_at = NOW()
                WHERE employee_id = :eid
            """), {"eid": row[0]})
        
        db.commit()
    except Exception as e:
        logger.error(f"HR action logging failed: {e}")
        db.rollback()
    finally:
        db.close()

async def alert_hr_team_identified(trigger_type: str, employee_name: str, 
                                    department: str, trigger_id: int, chat_id: int):
    """
    Send IDENTIFIED alert to HR with employee name and action buttons.
    This is for RED triggers and threshold crossings only.
    """
    hr_chat_id = os.getenv("PULSE_HR_CHAT_ID") or os.getenv("HUNT_TG_SUPERGROUP_ID")
    if not hr_chat_id:
        logger.warning("No HR chat ID configured for pulse alerts")
        return
    
    # Get risk score
    db = SessionLocal()
    try:
        row = db.execute(text(
            "SELECT current_score FROM pulse_risk_scores WHERE employee_id = :eid"
        ), {"eid": chat_id}).fetchone()
        risk_score = row[0] if row else 0
        
        # Get history
        history_rows = db.execute(text("""
            SELECT trigger_type, severity, risk_points, fired_at
            FROM pulse_triggers
            WHERE employee_id = :eid AND fired_at > NOW() - INTERVAL '30 days'
            ORDER BY fired_at DESC LIMIT 5
        """), {"eid": chat_id}).fetchall()
    finally:
        db.close()
    
    severity_emoji = "🔴" if trigger_type == "resignation" else "🟡"
    
    history_lines = []
    for row in history_rows:
        t_type, t_sev, t_pts, t_date = row
        date_str = t_date.strftime("%d.%m") if t_date else "?"
        sev_icon = "🔴" if t_sev == "red" else "🟡"
        history_lines.append(f"  {sev_icon} {date_str} — {t_type} (+{t_pts})")
    
    history_text = "\n".join(history_lines) if history_lines else "  (перший сигнал)"
    
    alert_text = (
        f"{severity_emoji} Team Pulse — Тривога\n\n"
        f"👤 {employee_name}\n"
        f"🏢 Відділ: {department or 'Невідомий'}\n"
        f"⚠️ Рівень ризику: {risk_score}/10\n"
        f"📌 Тригер: {trigger_type}\n\n"
        f"📊 Історія сигналів (30 днів):\n{history_text}"
    )
    
    # Send with action buttons
    keyboard = {
        "inline_keyboard": [[
            {"text": "✅ Взяв в роботу", "callback_data": f"hr_pulse:hr_action:reviewing:{trigger_id}"},
            {"text": "✔️ Вирішено", "callback_data": f"hr_pulse:hr_action:resolved:{trigger_id}"},
            {"text": "❌ Хибна тривога", "callback_data": f"hr_pulse:hr_action:false_positive:{trigger_id}"}
        ]]
    }
    
    await send_telegram_message_with_keyboard(hr_chat_id, alert_text, keyboard)

async def pulse_videos():
    db = SessionLocal()
    try:
        rows = db.execute(text("""
            SELECT video_id, COUNT(*) as views, 
                   COUNT(CASE WHEN completed THEN 1 END) as completed
            FROM pulse_video_views
            GROUP BY video_id
        """)).fetchall()
        
        total = sum(r[1] for r in rows)
        completed = sum(r[2] for r in rows)
        
        return {
            "total_views": total,
            "completion_rate": round(completed / total, 2) if total > 0 else 0,
            "by_video": [
                {"id": r[0], "views": r[1], "completion_rate": round(r[2]/r[1], 2) if r[1] > 0 else 0}
                for r in rows
            ]
        }
