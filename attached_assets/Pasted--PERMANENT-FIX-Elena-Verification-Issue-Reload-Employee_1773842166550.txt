def normalize_phone(phone_raw: str) -> tuple:
    """
    Extract and normalize phone numbers from Blitz format
    Returns: (phone_work_norm, phone_mobile_norm)
    """
    if pd.isna(phone_raw) or not phone_raw:
        return (None, None)
    
    phone_str = str(phone_raw)
    
    # Extract work phone (р) and mobile phone (л)
    work_phone = None
    mobile_phone = None
    
    # Pattern: (р) (XXX)XXX-XX-XX;(л) (XXX)XXX-XX-XX;
    if '(р)' in phone_str and '(л)' in phone_str:
        parts = phone_str.split(';')
        for part in parts:
            clean = re.sub(r'\D', '', part)  # Remove all non-digits
            if clean and len(clean) >= 9:
                if '(р)' in part and not work_phone:
                    work_phone = clean
                elif '(л)' in part and not mobile_phone:
                    mobile_phone = clean
    elif '(л)' in phone_str:
        # Only mobile
        clean = re.sub(r'\D', '', phone_str)
        if clean and len(clean) >= 9:
            mobile_phone = clean
    elif '(р)' in phone_str:
        # Only work
        clean = re.sub(r'\D', '', phone_str)
        if clean and len(clean) >= 9:
            work_phone = clean
    
    return (work_phone, mobile_phone)

async def reload_cache():
