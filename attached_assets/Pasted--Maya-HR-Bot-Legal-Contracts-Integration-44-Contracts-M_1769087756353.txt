from fastapi.responses import FileResponse
import os

@app.get("/api/legal-contract/{filename}")
async def download_legal_contract(filename: str):
    """Download legal contract template."""
    # Security: validate filename to prevent directory traversal
    if '..' in filename or '/' in filename:
        raise HTTPException(status_code=400, detail="Invalid filename")
    
    filepath = f"/home/app/legal_contracts/{filename}"

    return FileResponse(

        filename=filename
    )
