async def run_vacancy_posting(vacancy_id: int, chat_id: int, 
                               thread_id: int, db):
    """
    Post formatted vacancy to all active TG channels via Telethon.
    Uses HR_TELETHON_SESSION env var (Natalia's session).
    Falls back to TELETHON_SESSION if HR session not set.
    """
    
    # 1. Load vacancy from DB
    # 2. Format public vacancy post:
    
    POST_FORMAT = """
🔔 Вакансія | Торговий Дім АВ

💼 {position}
📍 {city}

📋 Вимоги:
{requirements_formatted}

💰 {salary_text}

📩 Відгукнутись: {contact}
#вакансія #{position_tag} #{city_tag}
"""
    
    contact = os.getenv("HR_CONTACT", "@avtd_hr")
    position_tag = position.replace(" ", "_").lower()
    city_tag = city.lower() if city else "україна"
    requirements_formatted = "\n".join([f"— {r}" for r in requirements])
    salary_text = f"До {salary_max}$/міс" if salary_max else "За домовленістю"
    
    # 3. Load active channels from hunt_sources
    # 4. Connect Telethon:
    
    session_string = os.getenv("HR_TELETHON_SESSION") or os.getenv("TELETHON_SESSION")
    api_id = int(os.getenv("TELETHON_API_ID", "0"))
    api_hash = os.getenv("TELETHON_API_HASH", "")
    
    if not all([session_string, api_id, api_hash]):
        logger.warning("Telethon credentials not configured for posting")
        await send_tg_message(chat_id, thread_id, 
            "⚠️ Credentials not configured. Set HR_TELETHON_SESSION in env vars.")
        return
    
    # 5. Post to each channel:
    posted = []
    failed = []
    
    async with TelegramClient(StringSession(session_string), api_id, api_hash) as client:
        for channel in channels:
            try:
                await client.send_message(channel, formatted_post)
                posted.append(channel)
                await asyncio.sleep(3)  # Rate limit - 3 seconds between posts
            except Exception as e:
                logger.error(f"Failed to post to {channel}: {e}")
                failed.append(channel)
    
    # 6. Update vacancy status to 'posted'
    # 7. Save posting results to new hunt_postings table
    
    # 8. Send report to TG thread:
    report = f"📢 Вакансію розміщено!\n\n"
    report += f"✅ Успішно: {len(posted)} каналів\n"
    if failed:
        report += f"❌ Помилка: {len(failed)} каналів\n"
    report += f"\nКанали: {', '.join(['@'+c for c in posted])}"
    
    await send_tg_message(chat_id, thread_id, report)

### New DB table (add to migration):

CREATE TABLE IF NOT EXISTS hunt_postings (
    id SERIAL PRIMARY KEY,
    vacancy_id INTEGER REFERENCES hunt_vacancies(id),
    channel VARCHAR(100),
