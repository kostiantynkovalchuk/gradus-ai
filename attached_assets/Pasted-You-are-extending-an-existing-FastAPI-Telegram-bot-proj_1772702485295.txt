### backend/services/hunt_service.py
Main orchestration service.

async def run_hunt(vacancy_id: int, vacancy_text: str, 
                   thread_id: int, chat_id: int, db):
    """
    1. Parse vacancy → update hunt_vacancies with parsed fields
    
    2. Send status message to TG thread:
       "🔍 Шукаю кандидатів...\n⏳ Це займе 1-2 хвилини"
       Save this message_id for potential editing later
    
    3. Get active sources from hunt_sources table
    
    4. Run scrapers in parallel:
       results = await asyncio.gather(
           search_tg_channels(vacancy, sources),
           search_workua(vacancy),
           return_exceptions=True
       )
       Flatten results, ignore exceptions (log them)
    
    5. If total results == 0:
       Send to thread: 
       "⚠️ Кандидатів не знайдено.\n💡 Спробуйте переформулювати вакансію або розширити вимоги."
       Update vacancy status to 'no_results'
       Return
    
    6. Score all candidates with Claude Haiku in parallel (max 10):
       scored = await asyncio.gather(*[
           score_candidate(c, vacancy) for c in results[:10]
       ])
    
    7. Merge score_result back into candidate dicts
    
    8. Sort by score descending, take top 5
    
    9. Save ALL scored candidates to hunt_candidates table
    
    10. Edit status message to:
        "✅ Знайдено {total} кандидатів. Показую топ {min(5,total)}:"
    
    11. For each top candidate (index 1-5):
        - Format card using format_candidate_card()
        - Send to TG thread with inline keyboard:
          [✅ В роботу] [❌ Пропустити] [💾 Зберегти]
          callback_data: 
            "hunt_approve_{candidate_id}"
            "hunt_reject_{candidate_id}"  
            "hunt_save_{candidate_id}"
        - Small delay between cards: await asyncio.sleep(0.5)
    
    12. Send final message to thread:
        "──────────────────\n🔄 Потрібно більше кандидатів?\nВідправте вакансію ще раз або уточніть вимоги."
    
    13. Update vacancy status to 'completed'
    """
