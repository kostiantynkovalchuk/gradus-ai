async def check_preset_answer(self, query: str) -> Optional[Dict]:
    """
    Check if query matches any preset answer
    Now supports multiple patterns per preset
    """
    query_lower = query.lower().strip()
    
    async with get_db() as db:
        # Get all active presets ordered by priority
        result = await db.execute(text("""
            SELECT 
                id, 
                question_pattern, 
                answer_text, 
                content_ids,
                priority
            FROM hr_preset_answers 
            WHERE is_active = true 
            ORDER BY priority DESC, id
        """))
        
        presets = result.fetchall()
    
    # Check each preset's patterns
    for preset in presets:
        pattern = preset[1].lower()
        
        # Exact match
        if pattern in query_lower or query_lower in pattern:
            await self._increment_preset_usage(preset[0])
            return {
                'answer': preset[2],
                'content_ids': preset[3],
                'source': 'preset',
                'match_type': 'exact'
            }
        
        # Fuzzy match (85% threshold)
        similarity = fuzz.ratio(query_lower, pattern)
        if similarity >= 85:
            await self._increment_preset_usage(preset[0])
            return {
                'answer': preset[2],
                'content_ids': preset[3],
                'source': 'preset',
                'match_type': 'fuzzy',
                'similarity': similarity
            }
    
    return None

async def _increment_preset_usage(self, preset_id: int):
    """Track preset usage for analytics"""
    async with get_db() as db:
        await db.execute(text("""
            UPDATE hr_preset_answers 
            SET usage_count = usage_count + 1 
            WHERE id = :id
        """), {'id': preset_id})
        await db.commit()

@router.get("/presets")
async def get_all_presets(
    category: str = None,
    active_only: bool = True
):
    """Get all preset answers, optionally filtered by category"""
    
    query = """
        SELECT 
            id,
            question_pattern,
            answer_text,
            content_ids,
            category,
            priority,
            usage_count,
            is_active
        FROM hr_preset_answers
        WHERE 1=1
    """
    
    params = {}
    
    if active_only:
        query += " AND is_active = true"
    
    if category:
        query += " AND category = :category"
        params['category'] = category
    
    query += " ORDER BY priority DESC, usage_count DESC"
    
    async with get_db() as db:
        result = await db.execute(text(query), params)
        presets = result.fetchall()
    
    return {
        'count': len(presets),
        'presets': [
            {
                'id': p[0],
                'pattern': p[1],
                'answer': p[2],
                'content_ids': p[3],
                'category': p[4],
                'priority': p[5],
                'usage_count': p[6],
                'is_active': p[7]
            }
            for p in presets
        ]
    }
