```python
from fastapi import APIRouter, Request
from fastapi.responses import Response
import os
import httpx
from services.avatar_personalities import get_avatar_response

router = APIRouter()

TELEGRAM_BOT_TOKEN = os.getenv("TELEGRAM_BOT_TOKEN")

@router.post("/webhook")
async def handle_telegram_webhook(request: Request):
    """Handle incoming Telegram messages"""
    
    data = await request.json()
    
    # Process message
    if "message" in data:
        await process_telegram_message(data["message"])
    
    return {"ok": True}

async def send_typing_action(chat_id: int):
    """Show typing indicator"""
    if not TELEGRAM_BOT_TOKEN:
        return
    
    try:
        async with httpx.AsyncClient(timeout=5.0) as client:
            await client.post(
                f"https://api.telegram.org/bot{TELEGRAM_BOT_TOKEN}/sendChatAction",
                json={
                    "chat_id": chat_id,
                    "action": "typing"
                }
            )
    except Exception as e:
        print(f"‚ö†Ô∏è Error sending typing action: {e}")


async def send_telegram_message(chat_id: int, text: str):
    """Send message to Telegram user"""
    if not TELEGRAM_BOT_TOKEN:
        print("‚ùå No Telegram bot token available")
        return
    
    try:
        # Telegram has 4096 char limit
        max_length = 4096
        
        if len(text) > max_length:
            messages = [text[i:i+max_length] for i in range(0, len(text), max_length)]
        else:
            messages = [text]
        
        async with httpx.AsyncClient(timeout=10.0) as client:
            for msg in messages:
                response = await client.post(
                    f"https://api.telegram.org/bot{TELEGRAM_BOT_TOKEN}/sendMessage",
                    json={
                        "chat_id": chat_id,
                        "text": msg,
                        "parse_mode": "Markdown"  # Support basic formatting
                    },
                    timeout=10.0
                )
                
                if response.status_code != 200:
                    print(f"‚ùå Telegram send error: {response.text}")
                    raise Exception(f"Telegram API error: {response.status_code}")
                
                print(f"‚úÖ Message sent to Telegram chat {chat_id}")
    
    except Exception as e:
        print(f"‚ùå Error sending Telegram message: {e}")
        raise
```
