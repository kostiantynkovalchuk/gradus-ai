```python
async def ingest_document(title: str, content: str, section: str, subsection: str):
    """
    Standard pipeline to add any document to Maya's knowledge base
    1. Insert into hr_content
    2. Generate embeddings
    3. Upload to Pinecone
    4. Verify
    """
    # Step 1: Insert to database
    content_id = await db.fetchval("""
        INSERT INTO hr_content (section, subsection, title, content, content_type)
        VALUES ($1, $2, $3, $4, 'document')
        RETURNING id
    """, section, subsection, title, content)
    
    # Step 2: Generate embedding
    response = await openai.embeddings.create(
        model="text-embedding-3-small",
        input=content[:8000]
    )
    embedding = response.data[0].embedding
    
    # Step 3: Store embedding in DB
    await db.execute("""
        INSERT INTO hr_embeddings (content_id, embedding)
        VALUES ($1, $2)
    """, content_id, embedding)
    
    # Step 4: Upload to Pinecone
    pinecone_index.upsert(
        vectors=[(f"hr_content_{content_id}", embedding, {
            "content_id": content_id,
            "title": title,
            "section": section,
            "content": content[:1000]
        })],
        namespace="hr_content"
    )
    
    logger.info(f"‚úÖ Ingested document: {title} (ID: {content_id})")
    return content_id
```
