def post_to_linkedin_task(self):
    """
    Post to LinkedIn at scheduled time
    Runs: Mon/Wed/Fri at 9:00 AM
    Selects oldest approved LinkedIn content
    """
    logger.info("ğŸ¤– [SCHEDULER] Posting to LinkedIn...")
    
    try:
        from services.linkedin_poster import LinkedInPoster
        from models import ContentQueue
        
        db = self._get_db_session()
        
        # Get oldest approved LinkedIn content (FIFO queue)
        article = db.query(ContentQueue).filter(
            ContentQueue.status == 'approved',
            ContentQueue.target_platform.in_(['linkedin', 'both']),
            ContentQueue.posted_at.is_(None)
        ).order_by(ContentQueue.created_at.asc()).first()
        
        if not article:
            logger.info("No approved LinkedIn content to post")
            db.close()
            return
        
        # Post to LinkedIn
        linkedin_poster = LinkedInPoster()
        
        content_data = {
            'title': article.translated_title or article.extra_metadata.get('title', ''),
            'text': article.translated_text or article.original_text,
            'image_url': article.image_url,
            'source_url': article.source_url
        }
        
        result = linkedin_poster.post_to_linkedin(content_data)
        
        if result.get('status') == 'success':
            article.status = 'posted'
            article.posted_at = datetime.now()
            if not article.extra_metadata:
                article.extra_metadata = {}
            article.extra_metadata['linkedin_post_url'] = result.get('post_url')
            article.extra_metadata['posted_platform'] = 'linkedin'
            
            logger.info(f"âœ… Posted to LinkedIn: Article {article.id}")
            
            # Send success notification
            self._send_posting_notification(article, 'linkedin', result.get('post_url'))
        else:
            logger.error(f"âŒ LinkedIn posting failed: {result.get('message')}")
        
        db.commit()
        db.close()
        
    except Exception as e:
        logger.error(f"âŒ [SCHEDULER] LinkedIn posting failed: {e}")
        import traceback
        logger.error(traceback.format_exc())
