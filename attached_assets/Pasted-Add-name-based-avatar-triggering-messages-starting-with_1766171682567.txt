```python
def detect_avatar_role(message: str) -> str:
    """
    Detect which avatar should respond based on message content.
    Priority: 1) Name prefix, 2) Topic keywords
    """
    message_lower = message.lower().strip()
    
    # NEW: Check if message starts with avatar name (highest priority)
    name_triggers = {
        'maya': ['maya', '–º–∞–π—è'],
        'alex': ['alex', '–∞–ª–µ–∫—Å']
    }
    
    # Check first word of message
    first_word = message_lower.split()[0] if message_lower else ''
    
    # Remove punctuation from first word (handle "Maya," or "–ú–∞–π—è:")
    first_word_clean = first_word.rstrip(',:!.?')
    
    for avatar, names in name_triggers.items():
        if first_word_clean in names:
            return avatar
    
    # EXISTING: Topic-based detection (fallback if no name prefix)
    maya_keywords = ['—Ç—Ä–µ–Ω–¥', '—Ä–∏–Ω–æ–∫', '–º–∞—Ä–∫–µ—Ç–∏–Ω–≥', '—Å—Ç—Ä–∞—Ç–µ–≥—ñ—è', '–±—Ä–µ–Ω–¥', 
                     '–Ω–æ–≤–∏–Ω–∏', '–±—ñ–∑–Ω–µ—Å', '–∞–Ω–∞–ª—ñ–∑', '–ø–æ–∑–∏—Ü—ñ–æ–Ω—É–≤–∞–Ω–Ω—è']
    alex_keywords = ['–∫–æ–∫—Ç–µ–π–ª—å', '—Ä–µ—Ü–µ–ø—Ç', '–±–∞—Ä', '–Ω–∞–ø—ñ–π', 'drink', 
                     '–º—ñ–∫—Å–æ–ª–æ–≥—ñ—è', '–º–∞—Ä–∂–∞', '—Å–æ–±—ñ–≤–∞—Ä—Ç—ñ—Å—Ç—å']
    
    if any(keyword in message_lower for keyword in maya_keywords):
        return 'maya'
    elif any(keyword in message_lower for keyword in alex_keywords):
        return 'alex'
    
    return 'general'
```
